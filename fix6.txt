handleNaturalInput 함수 상단에 마이크 종료 코드 추가.

현재 handleNaturalInput 함수 시작부분:

async function handleNaturalInput() {
  const inp   = document.getElementById('nlInput');
  let   raw   = inp.value.trim();
  const btn   = document.getElementById('nlSendBtn');
  const btnTxt = document.getElementById('nlBtnTxt');
  if (!raw) return;

변경 후:

async function handleNaturalInput() {
  // 마이크 활성 중이면 즉시 종료 (중복 입력 방지)
  if (_micActive) {
    _micShouldStop = true;
    if (_recognition) { try { _recognition.stop(); } catch(e) {} _recognition = null; }
    _stopMicUI();
  }
  const inp   = document.getElementById('nlInput');
  let   raw   = inp.value.trim();
  const btn   = document.getElementById('nlSendBtn');
  const btnTxt = document.getElementById('nlBtnTxt');
  if (!raw) return;

그리고 _stopMic 함수에서 handleNaturalInput 자동 호출 부분 삭제.

현재 _stopMic:
function _stopMic() {
  _micShouldStop = true;
  if (_recognition) {
    try { _recognition.stop(); } catch(e) {}
    _recognition = null;
  }
  _stopMicUI();
  const inp = document.getElementById('nlInput');
  if (inp && inp.value.trim()) {
    setTimeout(() => handleNaturalInput(), 300);
  }
}

변경 후:
function _stopMic() {
  _micShouldStop = true;
  if (_recognition) {
    try { _recognition.stop(); } catch(e) {}
    _recognition = null;
  }
  _stopMicUI();
}

이유: 카운트다운 0일 때 _stopMic이 handleNaturalInput을 호출하면, 사용자가 이미 입력 버튼을 눌렀을 때 중복 전송됨. 
카운트다운 종료 시 자동 전송은 _startMicCountdown 함수에서 처리.

그리고 _startMicCountdown 함수의 remaining <= 0 부분 수정:

현재:
    if (remaining <= 0) {
      _micShouldStop = true;
      _stopMic();
      return;
    }

변경 후:
    if (remaining <= 0) {
      _micShouldStop = true;
      if (_recognition) { try { _recognition.stop(); } catch(e) {} _recognition = null; }
      _stopMicUI();
      // 카운트다운 종료 시 텍스트 있으면 자동 전송
      const inp = document.getElementById('nlInput');
      if (inp && inp.value.trim()) {
        setTimeout(() => handleNaturalInput(), 300);
      }
      return;
    }
