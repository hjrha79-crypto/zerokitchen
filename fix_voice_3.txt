// 하이브리드 음성인식: Web Speech(실시간 미리보기) + OpenAI(최종 정확 텍스트)
let _micActive = false;
let _micCountdown = null;
let _micShouldStop = false;
let _micTimeout = parseInt(localStorage.getItem('zk_mic_timeout') || '10');
let _micRemaining = 0;
let _micAccumulated = '';

// MediaRecorder (OpenAI 전송용)
let _mediaRecorder = null;
let _audioChunks = [];

// Web Speech API (실시간 미리보기용)
let _recognition = null;

function toggleMic() {
  localStorage.setItem('zk_mic_used', 'true');
  const hint = document.getElementById('micHint');
  if (hint) hint.remove();

  if (_micActive) {
    _micShouldStop = true;
    _stopAll();
    return;
  }

  // 마이크 권한 + 녹음 시작
  navigator.mediaDevices.getUserMedia({ audio: true })
    .then(stream => {
      _micShouldStop = false;
      _micAccumulated = document.getElementById('nlInput').value.trim();
      _micActive = true;
      _audioChunks = [];

      const btn = document.getElementById('micBtn');
      btn.classList.add('rec');

      // 1) MediaRecorder 시작 (OpenAI 전송용 녹음)
      _mediaRecorder = new MediaRecorder(stream, { mimeType: _getAudioMimeType() });
      _mediaRecorder.ondataavailable = (e) => {
        if (e.data.size > 0) _audioChunks.push(e.data);
      };
      _mediaRecorder.onstop = () => {
        stream.getTracks().forEach(t => t.stop());
        if (_audioChunks.length > 0) {
          _transcribeAudio();
        }
      };
      _mediaRecorder.start(500);

      // 2) Web Speech API 시작 (실시간 미리보기)
      _startPreview();

      // 3) 카운트다운 시작
      _startMicCountdown();

      // placeholder 변경
      const inp = document.getElementById('nlInput');
      inp.dataset.prevPlaceholder = inp.placeholder;
    })
    .catch(err => {
      showToast('마이크 접근 불가: ' + err.message);
    });
}

function _getAudioMimeType() {
  if (MediaRecorder.isTypeSupported('audio/webm;codecs=opus')) return 'audio/webm;codecs=opus';
  if (MediaRecorder.isTypeSupported('audio/webm')) return 'audio/webm';
  if (MediaRecorder.isTypeSupported('audio/mp4')) return 'audio/mp4';
  return 'audio/wav';
}

// Web Speech API — 실시간 미리보기
function _startPreview() {
  if (!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) return;
  const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  _recognition = new SR();
  _recognition.lang = 'ko-KR';
  _recognition.interimResults = true;
  _recognition.continuous = false;

  _recognition.onresult = (e) => {
    let final = '', interim = '';
    for (let i = 0; i < e.results.length; i++) {
      if (e.results[i].isFinal) final += e.results[i][0].transcript;
      else interim += e.results[i][0].transcript;
    }
    const inp = document.getElementById('nlInput');
    if (final) {
      _micAccumulated += (_micAccumulated ? ' ' : '') + final;
      inp.value = _micAccumulated;
    } else if (interim) {
      inp.value = _micAccumulated + (_micAccumulated && interim ? ' ' : '') + interim;
    }
  };

  _recognition.onend = () => {
    _recognition = null;
    // 아직 녹음 중이면 재시작 (브라우저 강제 종료 우회)
    if (_micActive && !_micShouldStop) {
      setTimeout(() => {
        if (_micActive && !_micShouldStop) _startPreview();
      }, 300);
    }
  };

  _recognition.onerror = (e) => {
    if (e.error === 'no-speech' || e.error === 'aborted') return;
    // 미리보기 오류는 무시 (OpenAI가 최종 처리)
  };

  try { _recognition.start(); } catch(e) {}
}

// OpenAI로 최종 변환
async function _transcribeAudio() {
  const inp = document.getElementById('nlInput');
  const btn = document.getElementById('nlSendBtn');
  const btnTxt = document.getElementById('nlBtnTxt');

  // 미리보기 텍스트 유지하면서 "변환 중..." 표시
  const previewText = inp.value;
  inp.placeholder = '변환 중...';
  btn.disabled = true;
  btnTxt.innerHTML = '<span class="spin"></span>';

  try {
    const mimeType = _audioChunks[0]?.type || 'audio/webm';
    const ext = mimeType.includes('mp4') ? 'mp4' : mimeType.includes('wav') ? 'wav' : 'webm';
    const blob = new Blob(_audioChunks, { type: mimeType });

    // 너무 짧은 녹음 무시
    if (blob.size < 5000) {
      _restoreUI(inp, btn, btnTxt);
      if (previewText.trim()) {
        // 미리보기 텍스트라도 있으면 그걸로 전송
        inp.value = previewText;
        setTimeout(() => handleNaturalInput(), 200);
      }
      return;
    }

    const formData = new FormData();
    formData.append('file', blob, 'recording.' + ext);

    const response = await fetch('/api/transcribe', {
      method: 'POST',
      body: formData
    });

    if (!response.ok) {
      const errData = await response.json().catch(() => ({}));
      throw new Error(errData.error?.message || 'API 오류 ' + response.status);
    }

    const data = await response.json();
    const text = (data.text || '').trim();

    if (text) {
      // OpenAI 결과로 교체 (미리보기보다 정확)
      inp.value = text;
      _restoreUI(inp, btn, btnTxt);
      setTimeout(() => handleNaturalInput(), 200);
    } else {
      // OpenAI 결과 없으면 미리보기 텍스트 사용
      _restoreUI(inp, btn, btnTxt);
      if (previewText.trim()) {
        inp.value = previewText;
        setTimeout(() => handleNaturalInput(), 200);
      } else {
        showToast('음성을 인식하지 못했습니다');
      }
    }
  } catch (e) {
    console.error('음성 변환 오류:', e);
    _restoreUI(inp, btn, btnTxt);
    // OpenAI 실패 시 미리보기 텍스트로 대체 전송
    if (previewText.trim()) {
      inp.value = previewText;
      showToast('정밀 변환 실패 — 미리보기 텍스트로 처리합니다');
      setTimeout(() => handleNaturalInput(), 200);
    } else {
      showToast('음성 변환 실패: ' + e.message);
    }
  }
}

function _restoreUI(inp, btn, btnTxt) {
  inp.placeholder = inp.dataset.prevPlaceholder || '말하거나 입력하세요';
  btn.disabled = false;
  btnTxt.textContent = '입력';
}

// 카운트다운
function _startMicCountdown() {
  const btn = document.getElementById('micBtn');
  _micRemaining = _micTimeout;
  _updateCountdownUI(btn);
  _micCountdown = setInterval(() => {
    _micRemaining--;
    if (_micRemaining <= 0) {
      _stopAll();
      return;
    }
    _updateCountdownUI(btn);
  }, 1000);
}

function _updateCountdownUI(btn) {
  btn.innerHTML = '<div class="mic-wave"><div class="mic-wave-bar"></div><div class="mic-wave-bar"></div><div class="mic-wave-bar"></div></div><span class="mic-countdown" style="position:absolute;bottom:-18px;font-size:11px;color:var(--red);font-weight:700">' + _micRemaining + '</span>';
}

// 전체 정지
function _stopAll() {
  _micShouldStop = true;
  // Web Speech 정지
  if (_recognition) {
    try { _recognition.stop(); } catch(e) {}
    _recognition = null;
  }
  // MediaRecorder 정지 (onstop에서 _transcribeAudio 호출됨)
  if (_mediaRecorder && _mediaRecorder.state !== 'inactive') {
    _mediaRecorder.stop();
  } else {
    _stopMicUI();
  }
}

function _stopMic() {
  _micShouldStop = true;
  _stopAll();
}

function _stopMicUI() {
  _micActive = false;
  const btn = document.getElementById('micBtn');
  btn.classList.remove('rec');
  btn.innerHTML = '🎤';
  if (_micCountdown) { clearInterval(_micCountdown); _micCountdown = null; }
  const inp = document.getElementById('nlInput');
  if (inp && inp.dataset.prevPlaceholder) {
    inp.placeholder = inp.dataset.prevPlaceholder;
  }
}

동작 흐름
━━━━━━━━━━━━━━━━━━━━━

마이크 탭
  → MediaRecorder 녹음 시작 (OpenAI용)
  → Web Speech API 시작 (실시간 미리보기)
  → 입력창에 실시간 텍스트 표시

녹음 중
  → 사용자가 말하면 입력창에 텍스트 실시간 표시 (Web Speech)
  → 동시에 음성 데이터 녹음 중 (MediaRecorder)

녹음 끝 (마이크 탭 or 카운트다운 0)
  → Web Speech 정지
  → MediaRecorder 정지 → OpenAI로 전송
  → "변환 중..." (1~2초)
  → 정확한 텍스트로 교체 → 자동 전송

OpenAI 실패 시
  → Web Speech 미리보기 텍스트로 대체 전송 (완전 실패 방지)
