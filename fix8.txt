_startMicCountdown 함수 전체 교체.

현재 _startMicCountdown 함수를 찾아서 전체 교체:

let _micRemaining = 0;

function _startMicCountdown() {
  const btn = document.getElementById('micBtn');
  _micRemaining = _micTimeout;
  btn.innerHTML = '<span class="mic-countdown">' + _micRemaining + '</span>';
  _micCountdown = setInterval(() => {
    _micRemaining--;
    if (_micRemaining <= 0) {
      _micShouldStop = true;
      if (_recognition) { try { _recognition.stop(); } catch(e) {} _recognition = null; }
      _stopMicUI();
      const inp = document.getElementById('nlInput');
      if (inp && inp.value.trim()) {
        setTimeout(() => handleNaturalInput(), 300);
      }
      return;
    }
    btn.innerHTML = '<span class="mic-countdown">' + _micRemaining + '</span>';
  }, 1000);
}

function _resetMicCountdown() {
  _micRemaining = _micTimeout;
  const btn = document.getElementById('micBtn');
  if (_micActive) btn.innerHTML = '<span class="mic-countdown">' + _micRemaining + '</span>';
}
