CREATE TABLE IF NOT EXISTS daily_orders (
  id              bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  store_id        bigint REFERENCES stores(store_id),
  order_date      date NOT NULL DEFAULT CURRENT_DATE,
  item_id         bigint REFERENCES items(item_id),
  item_name       text NOT NULL,
  recommended_qty numeric NOT NULL DEFAULT 0,
  final_qty       numeric NOT NULL DEFAULT 0,
  unit            text DEFAULT '개',
  vendor_id       bigint REFERENCES vendors(vendor_id),
  status          text NOT NULL DEFAULT 'pending'
                  CHECK (status IN ('pending','confirmed','deferred','cancelled')),
  source          text DEFAULT 'target_based'
                  CHECK (source IN ('target_based','consumption_based')),
  defer_count     integer DEFAULT 0,
  created_at      timestamptz DEFAULT now(),
  updated_at      timestamptz DEFAULT now()
);
ALTER TABLE daily_orders ENABLE ROW LEVEL SECURITY;
CREATE POLICY "daily_orders_all" ON daily_orders FOR ALL USING (true);

async function _deferAllOrders() {
  const inputs = document.querySelectorAll('.ord-stepper-val');
  let count = 0;
  let blocked = 0;
  const today = new Date().toISOString().slice(0, 10);

  for (const inp of inputs) {
    const qty = parseInt(inp.value) || 0;
    if (qty <= 0) continue;
    const itemId = parseInt(inp.dataset.itemId);
    const name = inp.dataset.name;
    const unit = inp.dataset.unit;
    const vendorId = inp.dataset.vendorId ? parseInt(inp.dataset.vendorId) : null;

    // 기존 이월 횟수 확인
    const { data: existingDefer } = await db.from('daily_orders')
      .select('defer_count')
      .eq('item_id', itemId).eq('status', 'deferred')
      .order('created_at', { ascending: false })
      .limit(1);
    const currentDefers = existingDefer?.[0]?.defer_count || 0;

    if (currentDefers >= 2) {
      blocked++;
      continue; // 이월 2회 제한
    }

    // 기존 deferred 있으면 업데이트, 없으면 새로 생성
    if (existingDefer?.length) {
      await db.from('daily_orders')
        .update({
          recommended_qty: qty, final_qty: qty,
          defer_count: currentDefers + 1,
          order_date: today,
          updated_at: new Date().toISOString()
        })
        .eq('item_id', itemId).eq('status', 'deferred');
    } else {
      await db.from('daily_orders').insert({
        store_id: SID, order_date: today,
        item_id: itemId, item_name: name,
        recommended_qty: qty, final_qty: qty,
        unit: unit, vendor_id: vendorId,
        status: 'deferred', source: 'target_based',
        defer_count: 1
      });
    }
    count++;
  }

  let msg = `${count}개 품목 내일로 이월됨`;
  if (blocked > 0) msg += ` (${blocked}개는 이월 2회 초과로 제외)`;
  showToast(msg, 3000);
  renderOrder();
}

기존 _confirmAllOrders 함수를 교체:

async function _confirmAllOrders() {
  const inputs = document.querySelectorAll('.ord-stepper-val');
  let count = 0;
  const today = new Date().toISOString().slice(0, 10);

  for (const inp of inputs) {
    const qty = parseInt(inp.value) || 0;
    if (qty <= 0) continue;
    const itemId = parseInt(inp.dataset.itemId);
    const name = inp.dataset.name;
    const unit = inp.dataset.unit;
    const vendorId = inp.dataset.vendorId ? parseInt(inp.dataset.vendorId) : null;

    // order_requests에 등록
    await db.from('order_requests').insert({
      store_id: SID, item_id: itemId, item_name: name,
      qty: qty, unit: unit, vendor_id: vendorId, status: 'pending'
    });

    // daily_orders에 기록
    await db.from('daily_orders').insert({
      store_id: SID, order_date: today,
      item_id: itemId, item_name: name,
      recommended_qty: parseInt(inp.defaultValue) || qty,
      final_qty: qty,
      unit: unit, vendor_id: vendorId,
      status: 'confirmed', source: 'target_based'
    });

    // 이월 상태 정리
    await db.from('daily_orders')
      .update({ status: 'cancelled', updated_at: new Date().toISOString() })
      .eq('item_id', itemId).eq('status', 'deferred');

    count++;
  }

  if (count) {
    showToast(`${count}개 품목 발주 등록됨`);
    renderOrder();
  } else {
    showToast('발주할 품목이 없습니다');
  }
}
