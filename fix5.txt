index.html 수정 1건 — 마이크 인식 안정화

_startRecognition 함수만 교체. 나머지 함수는 그대로 유지.

현재 _startRecognition 함수를 찾아서 전체 교체:

function _startRecognition() {
  const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  _recognition = new SR();
  _recognition.lang = 'ko-KR';
  _recognition.interimResults = true;
  _recognition.continuous = false;

  _recognition.onresult = (e) => {
    let final = '', interim = '';
    for (let i = 0; i < e.results.length; i++) {
      if (e.results[i].isFinal) final += e.results[i][0].transcript;
      else interim += e.results[i][0].transcript;
    }
    if (final) {
      _micAccumulated += (_micAccumulated ? ' ' : '') + final;
      document.getElementById('nlInput').value = _micAccumulated;
    } else {
      document.getElementById('nlInput').value = _micAccumulated + (_micAccumulated && interim ? ' ' : '') + interim;
    }
  };

  _recognition.onend = () => {
    _recognition = null;
    if (_micActive && !_micShouldStop) {
      setTimeout(() => {
        if (_micActive && !_micShouldStop) _startRecognition();
      }, 300);
      return;
    }
  };

  _recognition.onerror = (e) => {
    if (e.error === 'no-speech' || e.error === 'aborted') return;
    showToast('음성 인식 오류: ' + e.error);
    _micShouldStop = true;
    _stopMic();
  };

  try {
    _recognition.start();
  } catch(e) {}
}

변경 포인트 3가지:
1. continuous = false로 변경 (브라우저가 안정적으로 세션 관리)
2. final 텍스트를 _micAccumulated에 누적 (재시작해도 이전 인식 유지)
3. onend에서 재시작 딜레이 300ms (잔여 오디오 간섭 방지)

동작 흐름:
- "치즈 여섯봉지" 말하면 → final로 인식 → _micAccumulated에 저장
- 브라우저 세션 종료 → 300ms 대기 → 깨끗하게 재시작
- "브로콜리 두봉지" 이어서 말하면 → 추가 누적
- 카운트다운 0 또는 버튼 탭 → 전체 누적 텍스트 전송
