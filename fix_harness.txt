function _createDecisionEvent({ entity, ai_suggestion, final_value, user_decision, decision_action, edit_delta, harness_trigger }) {
  const event = {
    decision_id: crypto.randomUUID ? crypto.randomUUID() : Math.random().toString(36).slice(2),
    source_app: 'zero_kitchen',
    event_type: 'decision_made',
    entity,
    ai_suggestion,
    final_value,
    value_type: 'number',
    user_decision,
    decision_action,
    edit_delta,
    harness_trigger,
    reason: null,
    created_at: new Date().toISOString()
  };
  const key = 'zerokitchen_decision_events_v1';
  const existing = JSON.parse(localStorage.getItem(key) || '[]');
  existing.push(event);
  localStorage.setItem(key, JSON.stringify(existing));
  console.log('[DecisionEvent]', event);
  return event;
}

2. 버튼별 값 매핑 (절대 변경 금지)
━━━━━━━━━━━━━━━━━━━━━

수량 이상값 - 확인: user_decision=CONFIRM, decision_action=null, edit_delta=0
수량 이상값 - 수정: user_decision=EDIT, decision_action=null, edit_delta=null
재입력 - 합치기: user_decision=EDIT, decision_action=MERGE, edit_delta=current
재입력 - 새로 입력: user_decision=REJECT, decision_action=REPLACE, edit_delta=null
품목 매칭 - 기존 연결: user_decision=CONFIRM, decision_action=null
품목 매칭 - 새 품목: user_decision=REJECT, decision_action=null
